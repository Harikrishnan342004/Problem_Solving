import sys

def solve():
    data = sys.stdin.read().split()
    N = int(data[0])
    M = int(data[1])
    MOD = 10**9 + 7
    
    pairs = N // 2
    
    # Build valid first elements and their counts
    valid = {}
    for a in range(1, M + 1):
        if a * a > M:
            break
        cnt = M // a - a + 1
        if cnt > 0:
            valid[a] = cnt
    
    if not valid:
        print(0)
        return
    
    # DP: dp[a] = number of valid arrays ending with pair starting at a
    dp = {a: c for a, c in valid.items()}
    
    for _ in range(pairs - 1):
        total = sum(dp.values()) % MOD
        new_dp = {}
        for a, cnt in valid.items():
            prev = (total - dp.get(a, 0)) % MOD
            if prev > 0:
                new_dp[a] = prev * cnt % MOD
        dp = new_dp
        if not dp:
            print(0)
            return
    
    print(sum(dp.values()) % MOD)

solve()
